Como Estender o Projeto

1. Buscar Dados de uma API

Modificar a função fetchProducts() em products.js:

async function fetchProducts() { try { // Alterar para sua API const response = await fetch( 'https://api.seu-site.com/products' ); const data = await response.json(); return data; } catch (error) { console.error('Erro:', error); return PRODUCTS; // Fallback } }

2. Adicionar Filtro por Preço

No arquivo script.js, estenda a função filterProducts():

function filterProducts(filter = 'all', minPrice = 0, maxPrice = Infinity) { let filtered = appState.products; if (filter === 'bestseller') { filtered = filtered.filter(p => p.bestseller); } return filtered.filter(p => p.price >= minPrice && p.price <= maxPrice ); }

3. Implementar Busca por Texto

Adicione esta função em script.js:

function searchProducts(query) { const q = query.toLowerCase(); return appState.products.filter(p => p.name.toLowerCase().includes(q) || p.description.toLowerCase().includes(q) ); }

4. Adicionar Carrinho de Compras (LocalStorage)

Crie um arquivo cart.js:

const cart = { items: [], add(product) { const existing = this.items.find(p => p.id === product.id); if (existing) { existing.quantity += 1; } else { this.items.push({...product, quantity: 1}); } this.save(); }, save() { localStorage.setItem('cart', JSON.stringify(this.items)); }, load() { this.items = JSON.parse(localStorage.getItem('cart') || '[]'); } };

7. Modo Escuro (Dark Mode)

Adicione esta função em script.js:

function toggleDarkMode() { document.body.classList.toggle('dark-mode'); localStorage.setItem('darkMode', document.body.classList.contains('dark-mode') ); } // Ao carregar: if (localStorage.getItem('darkMode') === 'true') { document.body.classList.add('dark-mode'); }

8. Email de Notificação de Estoque

Chamar API backend em product-detail.js:

async function notifyOnStock(productId) { const email = prompt('Seu email:'); await fetch('https://api.seu-site.com/notify', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({productId, email}) }); }

10. SEO - Sitemap Dinâmico

Crie um arquivo sitemap.js:

function generateSitemap() { const baseUrl = 'https://seu-site.github.io/ecommerce'; let xml = '<?xml version="1.0" encoding="UTF-8"?>\n'; xml += '<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">\n'; xml += `<url><loc>${baseUrl}/index.html</loc></url>\n`; PRODUCTS.forEach(p => { xml += `<url><loc>${baseUrl}/product.html?id=${p.id}</loc></url>\n`; }); xml += '</urlset>'; return xml; }
